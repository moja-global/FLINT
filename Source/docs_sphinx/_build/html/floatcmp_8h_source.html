34 // This template class represents an IEEE floating-point number
35 // (either single-precision or double-precision, depending on the
36 // template parameters).
37 //
38 // The purpose of this class is to do more sophisticated number
39 // comparison. (Due to round-off error, etc, it's very unlikely that
40 // two floating-points will be equal exactly. Hence a naive
41 // comparison by the == operation often doesn't work.)
42 //
43 // Format of IEEE floating-point:
44 //
45 // The most-significant bit being the leftmost, an IEEE
46 // floating-point looks like
47 //
48 // sign_bit exponent_bits fraction_bits
49 //
50 // Here, sign_bit is a single bit that designates the sign of the
51 // number.
52 //
53 // For float, there are 8 exponent bits and 23 fraction bits.
54 //
55 // For double, there are 11 exponent bits and 52 fraction bits.
56 //
57 // More details can be found at
58 // http://en.wikipedia.org/wiki/IEEE_floating-point_standard.
59 //
60 // Template parameter:
61 //
62 // RawType: the raw floating-point type (either float or double)
63 template <typename RawType>
64 class FloatingPoint {
65  public:
66  // Defines the unsigned integer type that has the same size as the
67  // floating point number.
68  typedef typename TypeWithSize<sizeof(RawType)>::UInt Bits;
69 
70  // Constants.
71 
72  // # of bits in a number.
73  static const size_t kBitCount = 8 * sizeof(RawType);
74 
75  // # of fraction bits in a number.
76  static const size_t kFractionBitCount = std::numeric_limits<RawType>::digits - 1;
77 
78  // # of exponent bits in a number.
79  static const size_t kExponentBitCount = kBitCount - 1 - kFractionBitCount;
80 
81  // The mask for the sign bit.
82  static const Bits kSignBitMask = static_cast<Bits>(1) << (kBitCount - 1);
83 
84  // The mask for the fraction bits.
85  static const Bits kFractionBitMask = ~static_cast<Bits>(0) >> (kExponentBitCount + 1);
86 
87  // The mask for the exponent bits.
88  static const Bits kExponentBitMask = ~(kSignBitMask | kFractionBitMask);
89 
90  // How many ULP's (Units in the Last Place) we want to tolerate when
91  // comparing two numbers. The larger the value, the more error we
92  // allow. A 0 value means that two numbers must be exactly the same
93  // to be considered equal.
94  //
95  // The maximum error of a single floating-point operation is 0.5
96  // units in the last place. On Intel CPU's, all floating-point
97  // calculations are done with 80-bit precision, while double has 64
98  // bits. Therefore, 4 should be enough for ordinary use.
99  //
100  // See the following article for more details on ULP:
101  // http://randomascii.wordpress.com/2012/02/25/comparing-floating-point-numbers-2012-edition/
102  static const size_t kMaxUlps = 4;
103 
104  // Constructs a FloatingPoint from a raw floating-point number.
105  //
106  // On an Intel CPU, passing a non-normalized NAN (Not a Number)
107  // around may change its bits, although the new value is guaranteed
108  // to be also a NAN. Therefore, don't expect this constructor to
109  // preserve the bits in x when x is a NAN.
110  explicit FloatingPoint(const RawType& x) { u_.value_ = x; }
111 
112  // Static methods
113 
114  // Reinterprets a bit pattern as a floating-point number.
115  //
116  // This function is needed to test the AlmostEquals() method.
117  static RawType ReinterpretBits(const Bits bits) {
118  FloatingPoint fp(0);
119  fp.u_.bits_ = bits;
120  return fp.u_.value_;
121  }
122 
123  // Returns the floating-point number that represent positive infinity.
124  static RawType Infinity() { return ReinterpretBits(kExponentBitMask); }
125 
126  // Returns the maximum representable finite floating-point number.
127  static RawType Max() { return std::numeric_limits<RawType>::max(); }
128 
129  // Non-static methods
130 
131  // Returns the bits that represents this number.
132  const Bits& bits() const { return u_.bits_; }
133 
134  // Returns the exponent bits of this number.
135  Bits exponent_bits() const { return kExponentBitMask & u_.bits_; }
136 
137  // Returns the fraction bits of this number.
138  Bits fraction_bits() const { return kFractionBitMask & u_.bits_; }
139 
140  // Returns the sign bit of this number.
141  Bits sign_bit() const { return kSignBitMask & u_.bits_; }
142 
143  // Returns true iff this is NAN (not a number).
144  bool is_nan() const {
145  // It's a NAN if the exponent bits are all ones and the fraction
146  // bits are not entirely zeros.
147  return (exponent_bits() == kExponentBitMask) && (fraction_bits() != 0);
148  }
149 
150  // Returns true iff this number is at most kMaxUlps ULP's away from
151  // rhs. In particular, this function:
152  //
153  // - returns false if either number is (or both are) NAN.
154  // - treats really large numbers as almost equal to infinity.
155  // - thinks +0.0 and -0.0 are 0 DLP's apart.
156  bool AlmostEquals(const FloatingPoint& rhs) const {
157  // The IEEE standard says that any comparison operation involving
158  // a NAN must return false.
159  if (is_nan() || rhs.is_nan()) return false;
160 
161  return DistanceBetweenSignAndMagnitudeNumbers(u_.bits_, rhs.u_.bits_) <= kMaxUlps;
162  }
163 
164  private:
165  // The data type used to store the actual floating-point number.
166  union FloatingPointUnion {
167  RawType value_; // The raw floating-point number.
168  Bits bits_; // The bits that represent the number.
169  };
170 
171  // Converts an integer from the sign-and-magnitude representation to
172  // the biased representation. More precisely, let N be 2 to the
173  // power of (kBitCount - 1), an integer x is represented by the
174  // unsigned number x + N.
175  //
176  // For instance,
177  //
178  // -N + 1 (the most negative number representable using
179  // sign-and-magnitude) is represented by 1;
180  // 0 is represented by N; and
181  // N - 1 (the biggest number representable using
182  // sign-and-magnitude) is represented by 2N - 1.
183  //
184  // Read http://en.wikipedia.org/wiki/Signed_number_representations
185  // for more details on signed number representations.
186  static Bits SignAndMagnitudeToBiased(const Bits& sam) {
187  if (kSignBitMask & sam) {
188  // sam represents a negative number.
189  return ~sam + 1;
190  } else {
191  // sam represents a positive number.
192  return kSignBitMask | sam;
193  }
194  }
195 
196  // Given two numbers in the sign-and-magnitude representation,
197  // returns the distance between them as an unsigned number.
198  static Bits DistanceBetweenSignAndMagnitudeNumbers(const Bits& sam1, const Bits& sam2) {
199  const Bits biased1 = SignAndMagnitudeToBiased(sam1);
200  const Bits biased2 = SignAndMagnitudeToBiased(sam2);
201  return (biased1 >= biased2) ? (biased1 - biased2) : (biased2 - biased1);
202  }
