54 inline double Flux::calculate(std::shared_ptr<IOperationResult> operationResult) {
55  if (!isMatchingFluxSource(operationResult)) {
56  return 0.0;

58 
59  double sum = 0.0;
60  for (auto it : operationResult->operationResultFluxCollection()) {
61  if (isIncluded(*it)) {
62  sum += it->value();
63  }
64  }
65 
66  return _subtract ? -sum : sum;
