85 inline int Timing::nSteps() const {
86  switch (_stepping) {
87  case TimeStepping::Daily: {
88  auto span = _startDate - _endDate;
89  return span.days() + 1;
90  }
91  case TimeStepping::Annual:
92  return _endDate.year() - _startDate.year() + 1;
93  case TimeStepping::Monthly:
94  return 1 + (_endDate.year() - _startDate.year()) * 12 + _endDate.month() - _startDate.month() + 1;
95  }
96  return 0;
97 }
